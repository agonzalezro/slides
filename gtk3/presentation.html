Differences

Python is OO, but C no. Did you know that? :)

1 b = Gtk.Button(...)
2 b.set_label("foo")

It changes, but continue being similar:

1 GtkWidget* b = gtk_button_new(...);
2 gtk_button_set_label(b, "foo");

 2

Ok, let's bind signals

 1 def bind_signals(self):
 2     self.window.connect('delete-event', Gtk.main_quit)
 3     self.button.connect('clicked', self.on_button_clicked)
 4     self.view.connect('load-progress-changed', self.on_load_progress)
 5 
 6 def get_full_url(self, text):
 7     return text if text.startswith('http') else 'http://%s' % text
 8 
 9 def on_button_clicked(self, widget):
10     self.view.open(
11             self.get_full_url(self.entry.get_text()))
12 
13 def on_load_progress(self, widget, amount):
14     self.progress.set_fraction(amount / 100.0)
15     if amount == 100 or amount == 0:
16         self.progress.hide()
17     else:
18         self.progress.show()

The twitter part

This is easy, isn't? Only take a look over it:

 1 URL_IDS = ('https://api.twitter.com/1/followers/ids.json?'
 2            'cursor=-1&screen_name=agonzalezro')
 3 URL_LOOKUP = ('https://api.twitter.com/1/users/lookup.json?'
 4               'user_id=%s&include_entities=true')
 5 
 6 def get_twitter_users(limit):
 7     response = urllib.urlopen(URL_IDS)
 8     ids = json.loads(response.read()).get('ids')
 9     ids = ','.join(map(str, ids[0:limit]))  # Max 100 per query
10     response = urllib.urlopen(URL_LOOKUP % ids)
11     users = json.loads(response.read())
12     minimal_info = []
13     for user in users:
14         minimal_info.append((user['profile_image_url'],
15                              '@%s' % user['screen_name']))
16     return minimal_info

1 def get_icon_from_url(url):

4     icon.write(urllib.urlopen(url).read())
5     icon.close()
6     return icon.get_pixbuf()
